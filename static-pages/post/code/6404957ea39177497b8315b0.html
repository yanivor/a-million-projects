export const getProcessedItems = (state, entity) => {
const { rows, table, table: { limit, page, sort, q }} = state[entity]
const [rowIds, count] = getRowIds(rows, limit, page, sort, q)

return ({
...state,
[entity]: {
...state[entity],
table: {
...table,
rowIds,
pagesCount:
countPages(count, limit),
count,
}
}
})
}
