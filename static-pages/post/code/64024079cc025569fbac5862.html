from __future__ import division
import serial
from NeuroPy import NeuroPy
import time

ser = serial.Serial('/dev/ttyACM1', 115200)

neuroPy = NeuroPy("/dev/rfcomm0")
neuroPy.start()

while True:
meditation = neuroPy.meditation
attention = neuroPy.attention

data = "x" + str(float(meditation/100)) + " y" + str(float(attention/100)) + "\r\n"
ser.write(data.encode())

print "gcode data to cnc shield - grbl server: ", data

time.sleep(1)
